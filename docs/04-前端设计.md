**交互逻辑：**
```javascript
Page({
  data: {
    goods: {},
    selectedSku: null,
    quantity: 1
  },

  onLoad(options) {
    const goodsId = options.id;
    this.loadGoodsDetail(goodsId);
  },

  // 选择规格
  onSkuSelect(e) {
    const sku = e.currentTarget.dataset.sku;
    this.setData({ selectedSku: sku });
  },

  // 数量变化
  onQuantityChange(e) {
    this.setData({ quantity: e.value });
  },

  // 加入购物车
  addToCart() {
    if (!this.data.selectedSku) {
      wx.showToast({ title: '请选择规格', icon: 'none' });
      return;
    }
    services.cart.add({
      skuId: this.data.selectedSku.id,
      quantity: this.data.quantity
    }).then(() => {
      wx.showToast({ title: '已加入购物车' });
    });
  },

  // 立即购买
  buyNow() {
    if (!this.data.selectedSku) {
      wx.showToast({ title: '请选择规格', icon: 'none' });
      return;
    }
    wx.navigateTo({
      url: `/pages/order/order-confirm?skuId=${this.data.selectedSku.id}&quantity=${this.data.quantity}`
    });
  }
});
```

**交互逻辑：**
```javascript
Page({
  data: {
    cartItems: [],
    allSelected: false,
    totalPrice: 0,
    selectedCount: 0,
    hasSelected: false
  },

  onShow() {
    this.loadCart();
  },

  // 加载购物车
  loadCart() {
    services.cart.list().then(res => {
      this.setData({ cartItems: res.data });
      this.calculateTotal();
    });
  },

  // 单选
  onItemSelect(e) {
    const id = e.currentTarget.dataset.id;
    const cartItems = this.data.cartItems.map(item => {
      if (item.id === id) {
        item.selected = !item.selected;
      }
      return item;
    });
    this.setData({ cartItems });
    this.calculateTotal();
  },

  // 全选
  onSelectAll(e) {
    const selected = e.detail.checked;
    const cartItems = this.data.cartItems.map(item => {
      item.selected = selected;
      return item;
    });
    this.setData({ cartItems, allSelected: selected });
    this.calculateTotal();
  },

  // 计算总价
  calculateTotal() {
    const selectedItems = this.data.cartItems.filter(item => item.selected);
    const totalPrice = selectedItems.reduce((sum, item) => {
      return sum + item.sku.price * item.quantity;
    }, 0);
    this.setData({
      totalPrice,
      selectedCount: selectedItems.length,
      hasSelected: selectedItems.length > 0
    });
  },

  // 数量变化
  onQuantityChange(e) {
    const id = e.currentTarget.dataset.id;
    const quantity = e.detail.value;
    services.cart.updateItem(id, { quantity }).then(() => {
      this.loadCart();
    });
  },

  // 去结算
  goToCheckout() {
    const selectedIds = this.data.cartItems
      .filter(item => item.selected)
      .map(item => item.id);
    wx.navigateTo({
      url: `/pages/order/order-confirm?cartItems=${JSON.stringify(selectedIds)}`
    });
  }
});
```

```html
      </view>
```

```html
          variant="light"
          theme="primary"

          data-keyword="{{item}}"
        >
          {{item}}
        </t-tag>
      </view>

```

**交互逻辑：**
```javascript
Page({
  data: {
    keyword: '',
    suggestions: [],
    searchHistory: [],
    hotKeywords: []
  },

  onLoad() {
    this.loadSearchHistory();
    this.loadHotKeywords();
  },

  // 输入变化（搜索建议）
  onInputChange(e) {
    const keyword = e.detail.value;
    if (keyword) {
      services.goods.getSuggestions(keyword).then(res => {
        this.setData({ suggestions: res.data });
      });
    } else {
      this.setData({ suggestions: [] });
    }
  },

  // 搜索
  onSearch(e) {
    const keyword = e.detail.value;
    if (!keyword) return;

    // 保存搜索历史
    this.saveSearchHistory(keyword);

    // 跳转到搜索结果
    wx.navigateTo({
      url: `/pages/goods/result?keyword=${keyword}`
    });
  },

  // 保存搜索历史
  saveSearchHistory(keyword) {
    let history = this.data.searchHistory;
    history = history.filter(item => item !== keyword);
    history.unshift(keyword);
    history = history.slice(0, 10); // 最多保存10条
    this.setData({ searchHistory: history });
    wx.setStorageSync('searchHistory', history);
  }
});
```

## 5. 状态管理

### 5.1 全局状态设计

使用 `app.js` 的 `globalData` 管理全局状态：

```javascript
// app.js
App({
  globalData: {
    // 用户信息
    userInfo: null,
    token: null,

    // 购物车数量
    cartCount: 0,

    // 系统配置
    config: {
      apiBaseUrl: 'https://api.example.com',
      uploadUrl: 'https://upload.example.com'
    }
  },

  // 更新用户信息
  updateUserInfo(userInfo) {
    this.globalData.userInfo = userInfo;
    this.globalData.token = userInfo.token;

    // 持久化存储
    wx.setStorageSync('userInfo', userInfo);
    wx.setStorageSync('token', userInfo.token);
  },

  // 清除用户信息
  clearUserInfo() {
    this.globalData.userInfo = null;
    this.globalData.token = null;
    wx.removeStorageSync('userInfo');
    wx.removeStorageSync('token');
  },

  // 更新购物车数量
  updateCartCount(count) {
    this.globalData.cartCount = count;
  }
});
```

### 5.2 页面数据管理

每个页面使用 `data` 管理页面状态：

```javascript
Page({
  data: {
    // 页面数据
    list: [],
    loading: false,
    hasMore: true,
    page: 1,
    pageSize: 20
  },

  onLoad() {
    this.loadData();
  },

  loadData() {
    if (this.data.loading || !this.data.hasMore) return;

    this.setData({ loading: true });

    services.goods.list({
      page: this.data.page,
      pageSize: this.data.pageSize
    }).then(res => {
      this.setData({
        list: [...this.data.list, ...res.data.list],
        hasMore: res.data.hasMore,
        loading: false
      });
    });
  }
});
```

### 5.3 本地存储策略

```javascript
// utils/storage.js
const Storage = {
  // 设置
  set(key, value) {

      wx.setStorageSync(key, value);
    } catch (e) {
      console.error('Storage.set error', e);
    }
  },

  // 获取
  get(key) {
    try {
      return wx.getStorageSync(key);
    } catch (e) {

    }

  // 删除
  remove(key) {
    try {
      wx.removeStorageSync(key);

      console.error('Storage.remove error', e);

  },

  clear() {

  }
};

module.exports = Storage;
```

### 8.5 防抖节流

```javascript
// utils/index.js
function debounce(fn, delay = 300) {
  let timer = null;
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => {
      fn.apply(this, args);
    }, delay);
  };
}

function throttle(fn, delay = 300) {
  let last = 0;
  return function(...args) {
    const now = Date.now();
    if (now - last > delay) {
      last = now;
      fn.apply(this, args);
    }
  };
}

module.exports = {
  debounce,
  throttle
};
```

**使用示例：**
```javascript
import { debounce, throttle } from '../../utils/index';

Page({
  data: {
    keyword: ''
  },

  // 防抖：搜索输入
  onSearchInput: debounce(function(e) {
    const keyword = e.detail.value;
    this.searchGoods(keyword);
  }, 500),

  // 节流：滚动事件
  onScroll: throttle(function(e) {
    console.log('滚动位置', e.detail.scrollTop);
  }, 200)
});
```

### 9.4 缓存策略

```javascript
// 数据缓存
const CACHE_KEY = 'goods_list';
const CACHE_DURATION = 5 * 60 * 1000; // 5分钟

Page({
  async loadData() {
    // 读取缓存
    const cachedData = wx.getStorageSync(CACHE_KEY);
    if (cachedData && Date.now() - cachedData.timestamp < CACHE_DURATION) {
      this.setData({ list: cachedData.data });
      return;
    }

    // 请求数据
    const res = await services.goods.list();
    this.setData({ list: res.data });

    // 写入缓存
    wx.setStorageSync(CACHE_KEY, {
      data: res.data,
      timestamp: Date.now()
    });
  }
});
```
