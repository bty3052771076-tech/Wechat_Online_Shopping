) VALUES (
  1, 'йҺөе¬«жәҖ', 'PHONE', 1, 2, 1, NOW()
);

-- йҸҢгғЁоҮ—йҚ’е—ҷиў«йҸҚжҲҜзҙҷй–«жҺ‘з¶ҠйҸҢгғЁоҮ—й”ӣ?-- MySQL 8.0 жөЈи·Ёж•Өй–«жҺ‘з¶ҠCTE
WITH RECURSIVE category_tree AS (
  -- йҚ©иҷ№о”…йҸҢгғЁоҮ—й”ӣе ңз«ҙз»ҫСғеһҺз»«дјҷзҙҡ
  SELECT id, parent_id, category_name, category_code, 1 as level,
         CAST(category_name AS CHAR(200)) as path
  FROM categories
  WHERE parent_id IS NULL AND status = 1

  UNION ALL

  -- й–«жҺ‘з¶ҠйҸҢгғЁоҮ—й”ӣе қз“ҷйҚ’е—ҷиў«й”ӣ?  SELECT c.id, c.parent_id, c.category_name, c.category_code, ct.level + 1,
         CONCAT(ct.path, ' > ', c.category_name)
  FROM categories c
  INNER JOIN category_tree ct ON ct.id = c.parent_id
  WHERE c.status = 1
)
SELECT * FROM category_tree ORDER BY level, sort_order;
```

  AVG(amount) as avg_amount
FROM table_name
WHERE condition;
```
